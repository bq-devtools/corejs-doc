'use strict';
/* global define */
define(
    ['corejs/engine/validate'],
    function(validate) {

    var SilkRoadError = function(jqXHR) {
        validate.isGreaterThanOrEqual(jqXHR.status, 400,
            'SilkRoadError is expecting an server error response, but got a response with status: '+jqXHR.status);
        
        this.httpStatus = jqXHR.status;
        if (isJson(jqXHR)) {
            var jsonError = JSON.parse(jqXHR.responseText);
            this.error = jsonError.error;
            this.errorDescription = jsonError.errorDescription;
            this.errorUri = jsonError.errorUri;
        } else {
            this.error = 'unknown';
            this.errorDescription = jqXHR.responseText;
        }
    };
    /**
     * Factory method to create a SilkRoadError
     * @param  {XMLHTTPRequest} jqHXR The server error response
     * @return {XMLHTTPRequest}       A new instance of SilkRoadError
     */
    SilkRoadError.create = function(jqXHR) {
        return new SilkRoadError(jqXHR);
    };

    var isJson = function(jqXHR) {
        return jqXHR.getResponseHeader('Content-Type').indexOf('json') !== -1;
    };

    //A SilkRoadError is a kind of Error:
    SilkRoadError.prototype = Error.prototype;

    return SilkRoadError;

});
