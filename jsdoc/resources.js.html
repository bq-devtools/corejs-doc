'use strict';
/* global define */
define([
    'corejs/app',
    'jquery',
    'backbone',
    'corejs/engine/session',
    'corejs/engine/silkroad/resources'
], function(app, $, Backbone, session, driverResources) {


    // Expose module
    app.resources = driverResources;

    /**
     * BACKBONE.SILKROAD.RESOURCES PLUGIN
     */

    // Keep original function
    var _sync = Backbone.sync;

    /**
     * Translate this full exampe query to a SilkRoad Compliant QueryString:
     * var params = {
     *     resource: 'resource:entity',
     *     id: 'id',
     *     rel: 'relType',
     *     uri: 'relUri',
     *     data: 'object',
     *     query: [
     *         { $eq: {field: 'value'} },
     *         { $eq: {field2: 'value2'} },
     *         { $gt: {field: 'value'} },
     *         { $gte: {field: 'value'} },
     *         { $lt: {field: 'value'} },
     *         { $lte: {field: 'value'} },
     *         { $ne: {field: 'value'} },
     *         { $eq: {field: 'value'} },
     *         { $like: {field: 'value'} },
     *         { $in: {field: ['value1', 'value2']} },
     *         { $all: {field: ['value1', 'value2']} }
     *     ],
     *     page: { page: 0, size: 10 },
     *     sort: { field: 'asc' }
     * };
     * @param  {object} params
     * @return {string}
     */
    Backbone.sync = function(method, model, options) {

        if (options.resourceType) {
            
            var builder,
                isCollection = model instanceof Backbone.Collection,
                defaultReject = $.Deferred().reject({ status: 409 }),
                dfd;

            // Make res builder
            if (isCollection) {
                if (options.rel) {
                    builder = driverResources.relation(options.resourceType, options.rel);
                } else {
                    builder = driverResources.collection(options.resourceType);
                }
            } else {
                builder = driverResources.resource(options.resourceType, (model.get('id') || options.id));
            }

            switch (method) {
            case 'read':
                if (isCollection) {
                    dfd = builder.get(options).then(function(data) {
                        model.set(data);
                    });
                } else {
                    dfd = builder.get().then(function(data) {
                        model.clear({silent: true});
                        model.set(data);
                    });
                }
                break;
            case 'update':
                if (isCollection) {
                    if(options.rel) {
                        dfd = builder.add(options.uri);
                    } else {
                        dfd = builder.add(options.data);
                    }
                } else {
                    return defaultReject;
                }
                break;
            case 'patch':
                return defaultReject;
            case 'delete':
                if (!isCollection) {
                    dfd = builder.delete();
                } else {
                    return defaultReject;
                }
                break;
            default:
                return defaultReject;
            }

            model.trigger('request', model, dfd, options);

            return dfd;
        } else {
            return _sync(method, model, options);
        }

    };

    return driverResources;

});
